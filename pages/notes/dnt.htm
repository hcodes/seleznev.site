Уже сейчас можно встроить поддержку doNotTrack в установку кода счётчика Метрики.
Для этого необходимо создание объекта new Ya.Metrika обвернуть условием.

function doNotTrack() {
    return (navigator.msDoNotTrack == '1' || navigator.doNotTrack == 'yes') ? true : false;
}

if (!doNotTrack()) {
    w.yaCounterXXXXXX = new Ya.Metrika({id: XXXXXX});
}

Где XXXXXX — номер счётчика Метрики.

Также можно подсчитать у какого количества пользователей установлен флажок запрета отслеживания.
Добавим передачу параметров при инициализации объекта Ya.Metrika. 

function doNotTrack() {
    return (navigator.msDoNotTrack == '1' || navigator.doNotTrack == 'yes') ? true : false;
}

var ymParams = {
    doNotTrack: {}
};

if (doNotTrack()) {
    ymParams.doNotTrack.on = 1;
}
else {
    ymParams.doNotTrack.off = 1;
}

w.yaCounterXXXXXX = new Ya.Metrika({id: XXXXXX, params: ymParams});

Статистику можно будет посмотреть в отчёте «Параметры визитов».
